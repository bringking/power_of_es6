Arrow functions

A function shorthand or Lambda using the fat-arrow " => " syntax.

Not just syntax sugar. Arrow functions automatically bind "this" from the containing scope.

ES6

(function() { //array var items = [1,2,3,4]; //expression bodies or lambda var byTwo = items.map( (i) => i*2); //with a block var byFour = items.map( (i) => { return i * 4; }); //lexical this function Dog() { this.name = "Dog"; this.nickNames = ["Fluffy", "Scruffy", "Biscuit"] this.print = function () { return this.nickNames.map( (n) => { return this.name + " is nicknamed " + n; }); }; } });
